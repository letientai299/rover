import { useId } from 'react';
import { FiHeart } from 'react-icons/fi';
import Tooltip from './Tooltip';

export function Anchor({ place }) {
  const id = useId();

return <><code id={id} style={{
    display: 'flex',
    minHeight: 'var(--sizes-6)',
    width: '100%',
    textAlign: 'center',
    justifyContent: 'center',
    alignItems: 'center',
  }}>{place}</code>

    <Tooltip place={place} anchor={id} style={{
      display: 'flex',
      padding: 4,
      flexDirection: 'column'
    }}>

      <FiHeart size={60} strokeWidth={1} />
      I'm a long tooltip at <code>{place}</code>
    </Tooltip>

  </>
}

export function Box(props) {
  const { style, children, ...rest } = props;
  return (
    <div
      style={{
        width: '100%',
        display: 'flex',
        minHeight: `var(--sizes-8)`,
        ...style,
      }}
      {...rest}
    >
      {children}
    </div>
  );
}
