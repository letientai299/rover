import { FaDeleteLeft } from 'react-icons/fa6';
import Button from './Button.tsx';
import { FiFilePlus, FiFolder, FiPhoneCall, FiSun } from 'react-icons/fi';

export function toggleTheme(e) {
  const root = document.getElementById('root');
  let cls = root.className;
  console.log(cls)
  if (cls === '') {
    cls = 'dark'
  } else if (cls.includes('light')) {
    cls = cls.replace('light', 'dark')
  } else {
    cls = cls.replace('dark', 'light')
  }

  root.className = cls;
}


## Basic

Icon at `start` position:
<Button icon={FiSun} onClick={toggleTheme} variation={'outline'}>Toggle theme</Button>

Icon at `end` position:
<Button icon={FiSun} iconPosition={'end'} variation={'outline'} onClick={toggleTheme}>Toggle theme</Button>

Icon only: <Button icon={FiSun} variation={'outline'} onClick={toggleTheme}/>
